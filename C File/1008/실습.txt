2. 가감승제 연산 문제 자동 출제 
   랜덤하게 10개의 문제 생성하여 리턴하는 함수 구현
   아래 코드를 참조 할 것
   문제를 저장하는 구조체 배열 정의 할 것

//==================================================================================
#include <stdio.h>
	#include <conio.h>	// getch
	#include <stdlib.h>	// srand
	#include <time.h>	// time

	void main()
	{
		srand(time(NULL));	// <== 초기화 영역

		while(1)
		{
			int num, num1;
			char op;
			system("cls");
			

			num =  rand()%10;	// 0 ~ 9
			num1 = rand()%10;	// 0 ~ 9
			op   = rand()%4;	// 0 ~ 3  0 ==> '+'  1 => '-'
			system("pause");
		}
	}
//=====================================================================================
